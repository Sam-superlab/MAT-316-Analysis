\section*{Countability and cardinality}

\begin{definition}[\textbf{Countable Set}]
    A set $S$ is said to be countable if there exists a bijection between $S$ and $\N$
\end{definition}

\begin{definition}[\textbf{Cardinality}]
The cardinality of a set $S$ is the number of elements in $S$. Two sets $S$ and $T$ has same cardinality iff there exists a bijection between $S$ and $T$
\end{definition}

\begin{definition}[\textbf{Finite and Infinite set}]
    A set $S$ is called finite if there exists some natural number $n$ such that $S$ has same cardinality as the set $\{1,2,3,...,n\}$. A set is not finite is called infinite.
\end{definition}


\section*{Sequence}

\begin{definition}[Sequence]
    A sequence is a function $f$ from $\N$ to $\R$ s.t.
    \begin{equation*}
        n \xrightarrow{f} a_n
    \end{equation*}
    we denote the sequence as $\{a_n\}_{n=1}^\infty$
\end{definition}

\begin{definition}[Sequence Convergence]
    A sequence $\{a_n\}_{n=1}^\infty$ is called converge to $a$ iff 
    \begin{equation*}
        \forall \varepsilon >0, \exists N(\varepsilon) \in \N \  s.t.\ \forall n \ge N(\varepsilon) \ \text{we have} |a_n-a|\le \varepsilon
    \end{equation*}
\end{definition}

\begin{definition}[Diverge]
    A sequence $\{a_n\}_{n=1}^\infty$ is called diverge to $+\infty$ if $\forall M,\exists N,s.t.,n\ge N$ implies $a_n \ge M$.
    Similarly the diverge to $-\infty$ are defined.
\end{definition}
